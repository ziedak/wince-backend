1. PostgreSQL Schema (Business & Operational Data)

Run these DDL statements against your PostgreSQL database (via psql or migration tool). All tables include appropriate indexes and constraints.
-- ============================================================
-- 1. Stores (tenants)
-- ============================================================
CREATE TABLE stores (
    id SERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    api_key_hash TEXT NOT NULL UNIQUE,          -- bcrypt hash of API key
    domain TEXT,
    plan TEXT DEFAULT 'free',
    rate_limit INTEGER DEFAULT 1000,            -- requests per minute
    config JSONB DEFAULT '{}',
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_stores_api_key_hash ON stores(api_key_hash);

-- ============================================================
-- 2. Customers (end users / shoppers)
-- ============================================================
CREATE TABLE customers (
    id SERIAL PRIMARY KEY,
    store_id INTEGER NOT NULL REFERENCES stores(id) ON DELETE CASCADE,
    distinct_id TEXT NOT NULL,                  -- client-generated persistent ID
    email TEXT,
    email_hash TEXT,                            -- for cross-device linking (optional)
    phone TEXT,
    email_consent BOOLEAN DEFAULT false,
    sms_consent BOOLEAN DEFAULT false,
    lifetime_value NUMERIC DEFAULT 0,
    last_seen_at TIMESTAMPTZ,
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(store_id, distinct_id)
);

CREATE INDEX idx_customers_lookup ON customers(store_id, distinct_id);
CREATE INDEX idx_customers_email ON customers(email);

-- ============================================================
-- 3. Interventions (log of every intervention sent)
-- ============================================================
CREATE TABLE interventions (
    id SERIAL PRIMARY KEY,
    intervention_id UUID NOT NULL UNIQUE,
    session_id TEXT NOT NULL,
    store_id INTEGER NOT NULL REFERENCES stores(id),
    distinct_id TEXT,
    type TEXT NOT NULL,                         -- 'discount', 'free_shipping', 'urgency', 'email_reminder'
    value NUMERIC,                              -- e.g., discount percent
    discount_code TEXT,
    sent_at TIMESTAMPTZ DEFAULT NOW(),
    delivered BOOLEAN DEFAULT false,
    delivered_via TEXT,                         -- 'websocket', 'email', 'sms'
    converted BOOLEAN DEFAULT false,
    revenue_attributed NUMERIC,
    experiment_id TEXT,                         -- A/B test identifier
    variant TEXT,                               -- 'control', 'rule_based', 'ai_v1', 'ai_v2'
    decision_latency_ms INTEGER,
    inference_confidence NUMERIC,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_interventions_session ON interventions(session_id);
CREATE INDEX idx_interventions_sent_at ON interventions(sent_at);
CREATE INDEX idx_interventions_store ON interventions(store_id);

-- ============================================================
-- 4. Discount codes (generated by Decision Engine)
-- ============================================================
CREATE TABLE discount_codes (
    code TEXT PRIMARY KEY,
    store_id INTEGER NOT NULL REFERENCES stores(id),
    session_id TEXT,
    intervention_id UUID REFERENCES interventions(intervention_id),
    value NUMERIC,                              -- discount percent
    expires_at TIMESTAMPTZ NOT NULL,
    used_at TIMESTAMPTZ,
    used_in_order_id TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_discount_codes_expires ON discount_codes(expires_at) WHERE used_at IS NULL;

-- ============================================================
-- 5. Policy rules (cooldown, max discount, daily budget)
-- ============================================================
CREATE TABLE policy_rules (
    id SERIAL PRIMARY KEY,
    store_id INTEGER NOT NULL REFERENCES stores(id) ON DELETE CASCADE,
    rule_type TEXT NOT NULL,                    -- 'cooldown_minutes', 'max_discount_percent', 'daily_budget_limit'
    parameters JSONB NOT NULL,                  -- e.g., {"value": 20}
    enabled BOOLEAN DEFAULT true,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_policy_rules_store ON policy_rules(store_id);

-- ============================================================
-- 6. Admin users (dashboard access)
-- ============================================================
CREATE TABLE admin_users (
    id SERIAL PRIMARY KEY,
    email TEXT UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    role TEXT DEFAULT 'viewer',                 -- 'admin', 'viewer', 'analyst'
    store_ids INTEGER[] DEFAULT '{}',          -- which stores they can access (empty = all)
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ============================================================
-- 7. Store usage (for billing / quotas)
-- ============================================================
CREATE TABLE store_usage (
    store_id INTEGER NOT NULL REFERENCES stores(id),
    date DATE NOT NULL,
    events_count BIGINT DEFAULT 0,
    predictions_count BIGINT DEFAULT 0,
    notifications_sent BIGINT DEFAULT 0,
    PRIMARY KEY (store_id, date)
);

-- ============================================================
-- 8. Daily budget (source of truth for discount limits)
-- ============================================================
CREATE TABLE daily_budget (
    store_id INTEGER NOT NULL REFERENCES stores(id),
    date DATE NOT NULL,
    total_discount_given NUMERIC DEFAULT 0,
    PRIMARY KEY (store_id, date)
);

-- ============================================================
-- 9. Experiments (A/B testing)
-- ============================================================
CREATE TABLE experiments (
    id SERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    store_id INTEGER NOT NULL REFERENCES stores(id),
    start_time TIMESTAMPTZ NOT NULL,
    end_time TIMESTAMPTZ,
    variants JSONB NOT NULL,                    -- e.g., {"control": 0, "rule_based": 50, "ai_v1": 50}
    active BOOLEAN DEFAULT true,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ============================================================
-- 10. Processed events (idempotency fallback)
-- ============================================================
CREATE TABLE processed_events (
    event_id UUID PRIMARY KEY,
    processed_at TIMESTAMPTZ DEFAULT NOW()
);
-- Auto-vacuum to keep size small (TTL 7 days)
CREATE INDEX idx_processed_events_processed_at ON processed_events(processed_at);

-- ============================================================
-- 11. Audit logs (compliance)
-- ============================================================
CREATE TABLE audit_logs (
    id SERIAL PRIMARY KEY,
    admin_user_id INTEGER REFERENCES admin_users(id),
    action TEXT NOT NULL,                       -- e.g., 'UPDATE_POLICY', 'REGENERATE_API_KEY'
    target TEXT,                                -- e.g., 'store:123'
    details JSONB,
    ip_address INET,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ============================================================
-- Optional: PgBouncer-friendly functions
-- ============================================================
-- For atomic budget update (increment)
CREATE OR REPLACE FUNCTION increment_daily_budget(
    p_store_id INTEGER,
    p_date DATE,
    p_amount NUMERIC
) RETURNS NUMERIC AS $$
BEGIN
    INSERT INTO daily_budget (store_id, date, total_discount_given)
    VALUES (p_store_id, p_date, p_amount)
    ON CONFLICT (store_id, date) DO UPDATE
    SET total_discount_given = daily_budget.total_discount_given + p_amount
    RETURNING total_discount_given INTO p_amount;
    RETURN p_amount;
END;
$$ LANGUAGE plpgsql;


2. ClickHouse Schema (Event Analytics)

Run these DDL statements against your ClickHouse cluster.

-- ============================================================
-- 1. Events table (distributed, partitioned)
-- ============================================================
CREATE TABLE events_local ON CLUSTER default_cluster
(
    timestamp DateTime64(3) CODEC(Delta, ZSTD),
    event_type LowCardinality(String),
    session_id String,
    distinct_id String,
    store_id UInt32,
    customer_email String CODEC(ZSTD),
    cart_value Float64,
    rage_click_count UInt8,
    is_frustrated Bool,
    properties JSON,
    server_timestamp DateTime64(3) DEFAULT now64()
) ENGINE = ReplicatedMergeTree('/clickhouse/tables/{shard}/events_local', '{replica}')
PARTITION BY toYYYYMM(timestamp)
ORDER BY (store_id, event_type, timestamp)
TTL timestamp + INTERVAL 90 DAY
SETTINGS index_granularity = 8192;

-- Distributed table (for querying across shards)
CREATE TABLE events ON CLUSTER default_cluster
AS events_local
ENGINE = Distributed(default_cluster, default, events_local, rand());

-- ============================================================
-- 2. Materialized view for daily aggregates
-- ============================================================
CREATE MATERIALIZED VIEW daily_abandonment_stats_local ON CLUSTER default_cluster
ENGINE = SummingMergeTree()
ORDER BY (store_id, date)
AS SELECT
    toDate(timestamp) AS date,
    store_id,
    countIf(event_type = 'checkout_abandon') AS abandonments,
    countIf(event_type = 'purchase') AS purchases,
    sumIf(cart_value, event_type = 'purchase') AS recovered_revenue
FROM events_local
GROUP BY date, store_id;

-- Distributed view
CREATE TABLE daily_abandonment_stats ON CLUSTER default_cluster
AS daily_abandonment_stats_local
ENGINE = Distributed(default_cluster, default, daily_abandonment_stats_local, rand());

3. Usage Notes for Developers
PostgreSQL

    Connection pool: Use PgBouncer (port 6432) with transaction pooling.

    Idempotency: The processed_events table is only used as a fallback for Bloom filter false positives. It should be vacuumed regularly (auto-vacuum enabled).

    Daily budget updates: Use the stored function increment_daily_budget to avoid race conditions.

ClickHouse

    Partitioning: Monthly partitions (toYYYYMM). Keeps 90 days of raw events.

    Materialized view: Updates automatically; query daily_abandonment_stats for fast reports.

    Sharding: Assumes a cluster with default_cluster. For single node, replace ReplicatedMergeTree with MergeTree and remove ON CLUSTER clauses.

Index Strategy

    PostgreSQL: Primary keys and indexes already defined. Add BRIN indexes on timestamps for very large tables if needed.

    ClickHouse: The ORDER BY (store_id, event_type, timestamp) provides efficient filtering by store and event type.


These schemas are source of truth for all database interactions. Any changes must be versioned and applied via migration scripts (e.g., migrate up). Use them as the foundation for your Enrichment Service, Decision Engine, and Admin API.
